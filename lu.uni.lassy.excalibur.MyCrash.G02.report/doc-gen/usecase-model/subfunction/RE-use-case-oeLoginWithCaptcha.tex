\subsubsection{subfunction-oeLoginWithCaptcha}

\label{RE-use-case-oeLoginWithCaptcha}


To use by any actor that inherits actAuthenticate to authenticate the actor as not being a robot/machine.		  


\begin{usecase}
  \addheading{Use-Case Description}
  \addsingletwocolumnrow{Name}{oeLoginWithCaptcha}
  \addsingletwocolumnrow{Scope}{system}
  \addsingletwocolumnrow{Level}{subfunction}
  
\addrowheading{Parameters}
\addnumberedsinglerow{ALogin: dtLogin}{}
\addnumberedsinglerow{APassword: dtPassword}{}
\addnumberedsinglerow{ACaptcha: dtCaptcha}{}

\addrowheading{Primary actor(s)}
\addnumberedsinglerow{}{\msrcode{actAuthenticated[active]}}



\addrowheading{Goal(s) description}
\addsinglerow{To use by any actor that inherits actAuthenticate to authenticate the actor as not being a robot/machine.}

\addrowheading{Protocol condition(s)}
\addnumberedsinglerow{}{
the iCrash system has been deployed.
the user is not yet logged in.
the user also needs to authenticate with a captcha field.
}

\addrowheading{Pre-condition(s)}
\addnumberedsinglerow{}{
none
}

\addrowheading{Main post-condition(s)}
\addnumberedsinglerow{}{
the user will be logged in if the username and password matched, also the captcha needed to be correct.
the administrators of the system will be notified of the outcome of this action.
}

\addrowheading{Additional Information}
\addsinglerow{
none
}

\end{usecase}
